# --- Configuration ---
$url = "http://localhost:3000/offre.php?NUMOFFRE=13894"
$requestCount = 30
# ---------------------

# Create an array to hold all the background jobs
$jobs = @()

Write-Host "Starting $requestCount requests to $url..."

# Start all the requests as background jobs
1..$requestCount | ForEach-Object {
    $job = Start-Job -ScriptBlock { 
        param($targetUrl)
        Invoke-WebRequest -Uri $targetUrl 
    } -ArgumentList $url
    $jobs += $job
}

Write-Host "All jobs started. Waiting for them to complete..."

# Wait for all jobs to finish and then show their output
Wait-Job -Job $jobs | Receive-Job

Write-Host "Done. Check your counter file."
